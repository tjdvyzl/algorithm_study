#include <bits/stdc++.h>
using namespace std;

int main() {

	int arrr[3] = { 1,2,128 };

	for (int i = 1; i < (1 << 3); ++i) {

		cout << bitset<8>(i) << ' ' << bitset<8>(1 << 0) << " " <<
			bitset<8>(1 << 1) << " " << bitset<8>(1 << 2) << "\n";

		for (int j = 0; j < 3; ++j) {
			if (i & (1 << j)) {
				cout << arrr[j] << " ";
			}
		}
		cout << "\n";
	}
}
